"use client";

import { useState, useEffect } from "react";
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import { 
  Shield, 
  Users, 
  Clock, 
  Heart, 
  Play, 
  Home, 
  Search, 
  Menu, 
  User, 
  Settings, 
  Mail, 
  Bell, 
  Calendar,
  Lock,
  Activity,
  Phone,
  MessageCircle,
  AlertTriangle,
  Eye,
  EyeOff
} from "lucide-react";

// Types for our data
interface UserData {
  name: string;
  email: string;
  password: string;
}

interface DeviceInfo {
  name: string;
  number: string;
  status: "Active" | "Idle" | "Offline";
  lastActive: string;
  invisibility: "Enabled" | "Disabled";
  encryption: "End-to-End Encrypted";
  battery: number;
}

interface EmotionData {
  timestamp: string;
  mood: "Happy" | "Neutral" | "Stressed" | "Anxious" | "Sad";
  score: number;
  activity: string;
}

interface ChatMetadata {
  platform: string;
  messages: number;
  contacts: number;
  lastActivity: string;
  messagePreview: string;
}

interface Alert {
  id: number;
  type: "Emotional" | "Usage" | "Social" | "Security";
  message: string;
  time: string;
  severity: "low" | "medium" | "high";
  activity: string;
}

export default function ParentControlDashboard() {
  // Authentication states
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [showSignup, setShowSignup] = useState(true);
  const [userData, setUserData] = useState<UserData>({ name: "", email: "", password: "" });
  const [loginData, setLoginData] = useState({ email: "", password: "" });
  const [showPassword, setShowPassword] = useState(false);
  
  // App states
  const [activeTab, setActiveTab] = useState("dashboard");
  const [invisibilityEnabled, setInvisibilityEnabled] = useState(true);
  const [notificationsEnabled, setNotificationsEnabled] = useState(true);
  const [deviceInfo, setDeviceInfo] = useState<DeviceInfo>({
    name: "iPhone 14 Pro",
    number: "+1 (555) 123-4567",
    status: "Active",
    lastActive: "Just now",
    invisibility: "Enabled",
    encryption: "End-to-End Encrypted",
    battery: 78
  });
  const [emotionData, setEmotionData] = useState<EmotionData[]>([
    { timestamp: new Date().toISOString(), mood: "Happy", score: 85, activity: "Playing games" },
    { timestamp: new Date(Date.now() - 3600000).toISOString(), mood: "Neutral", score: 65, activity: "Browsing social media" },
    { timestamp: new Date(Date.now() - 7200000).toISOString(), mood: "Stressed", score: 40, activity: "School work" },
    { timestamp: new Date(Date.now() - 10800000).toISOString(), mood: "Happy", score: 80, activity: "Video call with friends" },
    { timestamp: new Date(Date.now() - 14400000).toISOString(), mood: "Anxious", score: 30, activity: "Texting unknown number" },
  ]);
  const [chatMetadata, setChatMetadata] = useState<ChatMetadata[]>([
    { platform: "WhatsApp", messages: 124, contacts: 15, lastActivity: "1 hour ago", messagePreview: "Hey, are you coming to the party?" },
    { platform: "Instagram", messages: 89, contacts: 22, lastActivity: "30 mins ago", messagePreview: "Check out my new post!" },
    { platform: "Snapchat", messages: 210, contacts: 8, lastActivity: "5 mins ago", messagePreview: "Streaks! ðŸ”¥" },
    { platform: "TikTok", messages: 42, contacts: 31, lastActivity: "2 mins ago", messagePreview: "New follower request" },
  ]);
  const [alerts, setAlerts] = useState<Alert[]>([
    { id: 1, type: "Emotional", message: "Detected signs of anxiety", time: "2 hours ago", severity: "medium", activity: "Texting unknown number" },
    { id: 2, type: "Usage", message: "Excessive screen time detected", time: "5 hours ago", severity: "low", activity: "Playing games for 4 hours" },
    { id: 3, type: "Social", message: "Unusual messaging pattern", time: "1 day ago", severity: "high", activity: "Contacted unknown person" },
  ]);
  const [newAlertCount, setNewAlertCount] = useState(3);

  // Handle signup
  const handleSignup = (e: React.FormEvent) => {
    e.preventDefault();
    // In a real app, this would call an API
    setIsAuthenticated(true);
    localStorage.setItem("user", JSON.stringify(userData));
  };

  // Handle login
  const handleLogin = (e: React.FormEvent) => {
    e.preventDefault();
    // In a real app, this would validate credentials
    const storedUser = localStorage.getItem("user");
    if (storedUser) {
      const user = JSON.parse(storedUser);
      if (user.email === loginData.email) {
        setIsAuthenticated(true);
        return;
      }
    }
    alert("Invalid credentials. Please try again.");
  };

  // Handle logout
  const handleLogout = () => {
    setIsAuthenticated(false);
    setLoginData({ email: "", password: "" });
  };

  // Simulate real-time data updates
  useEffect(() => {
    if (!isAuthenticated) return;

    const interval = setInterval(() => {
      // Update device status
      setDeviceInfo(prev => ({
        ...prev,
        lastActive: "Just now",
        battery: Math.max(10, prev.battery - Math.floor(Math.random() * 3))
      }));

      // Add new emotion data point
      const moods: EmotionData["mood"][] = ["Happy", "Neutral", "Stressed", "Anxious", "Sad"];
      const activities: EmotionData["activity"][] = [
        "Playing games",
        "Browsing social media",
        "Texting friends",
        "Watching videos",
        "Texting unknown number",
        "Late night usage"
      ];
      
      const newMood = moods[Math.floor(Math.random() * moods.length)];
      const newScore = Math.floor(Math.random() * 100);
      const newActivity = activities[Math.floor(Math.random() * activities.length)];
      
      setEmotionData(prev => {
        const newData = [...prev];
        newData.unshift({
          timestamp: new Date().toISOString(),
          mood: newMood,
          score: newScore,
          activity: newActivity
        });
        // Keep only last 10 data points
        return newData.slice(0, 10);
      });

      // Update chat metadata
      setChatMetadata(prev => 
        prev.map(platform => {
          const newMessages = platform.messages + Math.floor(Math.random() * 5);
          const activities = [
            "New message received",
            "Sticker sent",
            "Photo shared",
            "Voice message",
            "Group chat activity"
          ];
          
          return {
            ...platform,
            messages: newMessages,
            lastActivity: "Just now",
            messagePreview: activities[Math.floor(Math.random() * activities.length)]
          };
        })
      );

      // Randomly generate new alerts
      if (Math.random() > 0.7) {
        const alertTypes: Alert["type"][] = ["Emotional", "Usage", "Social", "Security"];
        const severities: Alert["severity"][] = ["low", "medium", "high"];
        const activities: Alert["activity"][] = [
          "Texting unknown number",
          "Late night usage",
          "Excessive social media",
          "Contacted stranger",
          "Unusual messaging pattern"
        ];
        
        const newAlert: Alert = {
          id: Date.now(),
          type: alertTypes[Math.floor(Math.random() * alertTypes.length)],
          message: "Potential concern detected",
          time: "Just now",
          severity: severities[Math.floor(Math.random() * severities.length)],
          activity: activities[Math.floor(Math.random() * activities.length)]
        };
        
        setAlerts(prev => [newAlert, ...prev.slice(0, 4)]);
        setNewAlertCount(prev => prev + 1);
      }
    }, 5000); // Update every 5 seconds

    return () => clearInterval(interval);
  }, [isAuthenticated]);

  const toggleInvisibility = () => {
    setInvisibilityEnabled(!invisibilityEnabled);
    setDeviceInfo(prev => ({
      ...prev,
      invisibility: !invisibilityEnabled ? "Enabled" : "Disabled"
    }));
  };

  const toggleNotifications = () => {
    setNotificationsEnabled(!notificationsEnabled);
  };

  const clearAlerts = () => {
    setAlerts([]);
    setNewAlertCount(0);
  };

  // Format time for display
  const formatTime = (timestamp: string) => {
    const date = new Date(timestamp);
    return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
  };

  // Get mood color
  const getMoodColor = (mood: string) => {
    switch(mood) {
      case "Happy": return "bg-green-500";
      case "Neutral": return "bg-blue-500";
      case "Stressed": return "bg-yellow-500";
      case "Anxious": return "bg-orange-500";
      case "Sad": return "bg-red-500";
      default: return "bg-gray-500";
    }
  };

  // If not authenticated, show signup/login
  if (!isAuthenticated) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 flex items-center justify-center p-4">
        <Card className="w-full max-w-md">
          <CardHeader className="text-center">
            <div className="flex justify-center mb-4">
              <Shield className="h-12 w-12 text-blue-600" />
            </div>
            <CardTitle className="text-2xl">
              {showSignup ? "Create Account" : "Welcome Back"}
            </CardTitle>
            <CardDescription>
              {showSignup 
                ? "Sign up to monitor your child's device" 
                : "Sign in to access your dashboard"}
            </CardDescription>
          </CardHeader>
          <CardContent>
            {showSignup ? (
              <form onSubmit={handleSignup} className="space-y-4">
                <div className="space-y-2">
                  <Label htmlFor="name">Full Name</Label>
                  <Input
                    id="name"
                    placeholder="Enter your name"
                    value={userData.name}
                    onChange={(e) => setUserData({...userData, name: e.target.value})}
                    required
                  />
                </div>
                <div className="space-y-2">
                  <Label htmlFor="email">Email</Label>
                  <Input
                    id="email"
                    type="email"
                    placeholder="Enter your email"
                    value={userData.email}
                    onChange={(e) => setUserData({...userData, email: e.target.value})}
                    required
                  />
                </div>
                <div className="space-y-2">
                  <Label htmlFor="password">Password</Label>
                  <div className="relative">
                    <Input
                      id="password"
                      type={showPassword ? "text" : "password"}
                      placeholder="Create a password"
                      value={userData.password}
                      onChange={(e) => setUserData({...userData, password: e.target.value})}
                      required
                    />
                    <Button
                      type="button"
                      variant="ghost"
                      size="icon"
                      className="absolute right-2 top-1/2 transform -translate-y-1/2"
                      onClick={() => setShowPassword(!showPassword)}
                    >
                      {showPassword ? <EyeOff className="h-4 w-4" /> : <Eye className="h-4 w-4" />}
                    </Button>
                  </div>
                </div>
                <Button type="submit" className="w-full">
                  Create Account
                </Button>
                <div className="text-center text-sm">
                  Already have an account?{" "}
                  <button 
                    type="button"
                    className="text-blue-600 hover:underline"
                    onClick={() => setShowSignup(false)}
                  >
                    Sign In
                  </button>
                </div>
              </form>
            ) : (
              <form onSubmit={handleLogin} className="space-y-4">
                <div className="space-y-2">
                  <Label htmlFor="login-email">Email</Label>
                  <Input
                    id="login-email"
                    type="email"
                    placeholder="Enter your email"
                    value={loginData.email}
                    onChange={(e) => setLoginData({...loginData, email: e.target.value})}
                    required
                  />
                </div>
                <div className="space-y-2">
                  <Label htmlFor="login-password">Password</Label>
                  <div className="relative">
                    <Input
                      id="login-password"
                      type={showPassword ? "text" : "password"}
                      placeholder="Enter your password"
                      value={loginData.password}
                      onChange={(e) => setLoginData({...loginData, password: e.target.value})}
                      required
                    />
                    <Button
                      type="button"
                      variant="ghost"
                      size="icon"
                      className="absolute right-2 top-1/2 transform -translate-y-1/2"
                      onClick={() => setShowPassword(!showPassword)}
                    >
                      {showPassword ? <EyeOff className="h-4 w-4" /> : <Eye className="h-4 w-4" />}
                    </Button>
                  </div>
                </div>
                <Button type="submit" className="w-full">
                  Sign In
                </Button>
                <div className="text-center text-sm">
                  Don't have an account?{" "}
                  <button 
                    type="button"
                    className="text-blue-600 hover:underline"
                    onClick={() => setShowSignup(true)}
                  >
                    Sign Up
                  </button>
                </div>
              </form>
            )}
          </CardContent>
        </Card>
      </div>
    );
  }

  // Main dashboard with tabs
  return (
    <div className="min-h-screen bg-gray-50">
      {/* Header */}
      <header className="bg-white border-b">
        <div className="px-4 py-3 sm:px-6 flex justify-between items-center">
          <div className="flex items-center">
            <Shield className="h-8 w-8 text-blue-600 mr-2" />
            <h1 className="text-xl font-bold text-gray-900">ParentGuard AI</h1>
            <Badge className="ml-3" variant="secondary">LIVE</Badge>
          </div>
          <div className="flex items-center space-x-4">
            <Button variant="ghost" size="icon" className="relative">
              <Bell className="h-5 w-5" />
              {newAlertCount > 0 && (
                <span className="absolute top-0 right-0 flex h-3 w-3">
                  <span className="animate-ping absolute inline-flex h-full w-full rounded-full bg-red-400 opacity-75"></span>
                  <span className="relative inline-flex rounded-full h-3 w-3 bg-red-500"></span>
                </span>
              )}
            </Button>
            <Button variant="ghost" size="icon">
              <Settings className="h-5 w-5" />
            </Button>
            <Button variant="ghost" size="icon" onClick={handleLogout}>
              <User className="h-5 w-5" />
            </Button>
          </div>
        </div>
      </header>

      <div className="flex">
        {/* Sidebar */}
        <nav className="w-64 bg-white border-r h-screen sticky top-0">
          <div className="p-4">
            <div className="flex items-center mb-8">
              <div className="bg-gray-200 border-2 border-dashed rounded-xl w-16 h-16" />
              <div className="ml-3">
                <p className="font-medium">{deviceInfo.name}</p>
                <div className="flex items-center">
                  <Activity className="h-3 w-3 text-green-500 mr-1" />
                  <p className="text-sm text-gray-500">Online</p>
                </div>
              </div>
            </div>
            
            <ul className="space-y-1">
              {[
                { id: "dashboard", label: "Dashboard", icon: Home },
                { id: "activity", label: "Activity Monitor", icon: Clock },
                { id: "emotions", label: "Emotional Health", icon: Heart },
                { id: "chats", label: "Chat Metadata", icon: Mail },
                { id: "settings", label: "Settings", icon: Settings },
              ].map((item) => {
                const Icon = item.icon;
                return (
                  <li key={item.id}>
                    <button
                      onClick={() => setActiveTab(item.id)}
                      className={`w-full flex items-center px-4 py-2 rounded-lg text-left ${
                        activeTab === item.id
                          ? "bg-blue-50 text-blue-600 font-medium"
                          : "text-gray-700 hover:bg-gray-100"
                      }`}
                    >
                      <Icon className="h-5 w-5 mr-3" />
                      {item.label}
                    </button>
                  </li>
                );
              })}
            </ul>
          </div>
        </nav>

        {/* Main Content */}
        <main className="flex-1 p-6">
          {/* Dashboard Tab */}
          {activeTab === "dashboard" && (
            <>
              <div className="mb-6">
                <h2 className="text-2xl font-bold text-gray-900">Dashboard</h2>
                <p className="text-gray-600">Monitor your child's device activity and wellbeing</p>
              </div>

              {/* Status Cards */}
              <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6 mb-6">
                <Card>
                  <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
                    <CardTitle className="text-sm font-medium">Device Status</CardTitle>
                    <Shield className="h-4 w-4 text-muted-foreground" />
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold flex items-center">
                      {deviceInfo.status}
                      <Activity className="h-4 w-4 text-green-500 ml-2" />
                    </div>
                    <p className="text-xs text-muted-foreground">Last active: {deviceInfo.lastActive}</p>
                  </CardContent>
                </Card>

                <Card>
                  <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
                    <CardTitle className="text-sm font-medium">Invisibility</CardTitle>
                    <Users className="h-4 w-4 text-muted-foreground" />
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold">{deviceInfo.invisibility}</div>
                    <p className="text-xs text-muted-foreground">Undetectable to child</p>
                  </CardContent>
                </Card>

                <Card>
                  <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
                    <CardTitle className="text-sm font-medium">Encryption</CardTitle>
                    <Lock className="h-4 w-4 text-muted-foreground" />
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold">{deviceInfo.encryption}</div>
                    <p className="text-xs text-muted-foreground">Metadata only logged</p>
                  </CardContent>
                </Card>

                <Card>
                  <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
                    <CardTitle className="text-sm font-medium">AI Alerts</CardTitle>
                    <Bell className="h-4 w-4 text-muted-foreground" />
                  </CardHeader>
                  <CardContent>
                    <div className="text-2xl font-bold">{alerts.length} Active</div>
                    <p className="text-xs text-muted-foreground">
                      {newAlertCount > 0 ? `${newAlertCount} new alerts` : "No new alerts"}
                    </p>
                  </CardContent>
                </Card>
              </div>

              {/* Device Info Card */}
              <Card className="mb-6">
                <CardHeader>
                  <CardTitle>Device Information</CardTitle>
                  <CardDescription>Current device status and details</CardDescription>
                </CardHeader>
                <CardContent>
                  <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                    <div className="flex items-center">
                      <Phone className="h-5 w-5 text-muted-foreground mr-2" />
                      <div>
                        <p className="text-sm text-muted-foreground">Phone Number</p>
                        <p className="font-medium">{deviceInfo.number}</p>
                      </div>
                    </div>
                    <div className="flex items-center">
                      <Shield className="h-5 w-5 text-muted-foreground mr-2" />
                      <div>
                        <p className="text-sm text-muted-foreground">Device Status</p>
                        <div className="flex items-center">
                          <Activity className="h-3 w-3 text-green-500 mr-1" />
                          <p className="font-medium">{deviceInfo.status}</p>
                        </div>
                      </div>
                    </div>
                    <div className="flex items-center">
                      <Battery className="h-5 w-5 text-muted-foreground mr-2" />
                      <div>
                        <p className="text-sm text-muted-foreground">Battery Level</p>
                        <div className="flex items-center">
                          <div className="w-24 bg-gray-200 rounded-full h-2 mr-2">
                            <div 
                              className="h-2 rounded-full bg-green-500" 
                              style={{ width: `${deviceInfo.battery}%` }}
                            ></div>
                          </div>
                          <span className="text-sm font-medium">{deviceInfo.battery}%</span>
                        </div>
                      </div>
                    </div>
                  </div>
                </CardContent>
              </Card>
            </>
          )}

          {/* Activity Monitor Tab */}
          {activeTab === "activity" && (
            <Card>
              <CardHeader>
                <CardTitle>Activity Monitor</CardTitle>
                <CardDescription>Track device usage patterns and screen time</CardDescription>
              </CardHeader>
              <CardContent>
                <div className="space-y-6">
                  <div>
                    <h3 className="font-medium mb-2">Screen Time Today</h3>
                    <div className="flex items-center">
                      <div className="w-32 bg-gray-200 rounded-full h-4 mr-4">
                        <div className="h-4 rounded-full bg-blue-500" style={{ width: "75%" }}></div>
                      </div>
                      <span className="text-lg font-medium">4h 32m</span>
                    </div>
                  </div>
                  
                  <div>
                    <h3 className="font-medium mb-2">App Usage</h3>
                    <div className="space-y-3">
                      {[
                        { app: "YouTube", time: "1h 45m", color: "bg-red-500" },
                        { app: "Instagram", time: "1h 22m", color: "bg-pink-500" },
                        { app: "TikTok", time: "58m", color: "bg-black" },
                        { app: "WhatsApp", time: "32m", color: "bg-green-500" },
                        { app: "Games", time: "25m", color: "bg-purple-500" },
                      ].map((item, index) => (
                        <div key={index} className="flex items-center">
                          <div className={`w-3 h-3 rounded-full ${item.color} mr-3`}></div>
                          <div className="flex-1">
                            <div className="flex justify-between text-sm">
                              <span>{item.app}</span>
                              <span>{item.time}</span>
                            </div>
                            <div className="w-full bg-gray-200 rounded-full h-2 mt-1">
                              <div 
                                className={`h-2 rounded-full ${item.color}`} 
                                style={{ width: `${100 - index * 15}%` }}
                              ></div>
                            </div>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                  
                  <div>
                    <h3 className="font-medium mb-2">Daily Usage Pattern</h3>
                    <div className="h-40 flex items-end space-x-1">
                      {[65, 45, 30, 50, 75, 90, 85, 70, 60, 55, 65, 80].map((value, index) => (
                        <div key={index} className="flex-1 flex flex-col items-center">
                          <div 
                            className="w-full bg-blue-500 rounded-t" 
                            style={{ height: `${value}%` }}
                          ></div>
                          <span className="text-xs mt-1 text-gray-500">
                            {index % 2 === 0 ? `${index + 8}am` : ''}
                          </span>
                        </div>
                      ))}
                    </div>
                  </div>
                </div>
              </CardContent>
            </Card>
          )}

          {/* Emotional Health Tab */}
          {activeTab === "emotions" && (
            <Card>
              <CardHeader>
                <CardTitle>Emotional Health Monitoring</CardTitle>
                <CardDescription>AI detected emotional patterns in real-time</CardDescription>
              </CardHeader>
              <CardContent>
                <div className="space-y-6">
                  <div>
                    <h3 className="font-medium mb-2">Current Mood</h3>
                    <div className="flex items-center">
                      <div className="w-16 h-16 rounded-full bg-orange-500 flex items-center justify-center mr-4">
                        <Heart className="h-8 w-8 text-white" />
                      </div>
                      <div>
                        <p className="text-2xl font-bold">Anxious</p>
                        <p className="text-gray-600">Detected 15 minutes ago</p>
                      </div>
                    </div>
                  </div>
                  
                  <div>
                    <h3 className="font-medium mb-2">Mood History</h3>
                    <div className="space-y-4">
                      {emotionData.map((data, index) => (
                        <div key={index} className="flex items-center border-b pb-3 last:border-0 last:pb-0">
                          <div className="w-24 text-sm text-gray-500">{formatTime(data.timestamp)}</div>
                          <div className="flex-1">
                            <div className="flex items-center justify-between mb-1">
                              <div className="flex items-center">
                                <div className={`w-3 h-3 rounded-full ${getMoodColor(data.mood)} mr-2`}></div>
                                <span className="text-sm font-medium">{data.mood}</span>
                              </div>
                              <span className="text-sm text-gray-500">{data.score}%</span>
                            </div>
                            <p className="text-xs text-gray-600">{data.activity}</p>
                            <div className="h-2 bg-gray-200 rounded-full mt-1">
                              <div 
                                className={`h-2 rounded-full ${getMoodColor(data.mood)}`} 
                                style={{ width: `${data.score}%` }}
                              ></div>
                            </div>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                  
                  <div>
                    <h3 className="font-medium mb-2">Weekly Summary</h3>
                    <div className="flex justify-between text-center">
                      {["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"].map((day, index) => (

                          <div className="mx-auto w-8 h-8 rounded-full bg-blue-100 flex items-center justify-center mb-1">
                            <span className="text-xs text-blue-800">{day}</span>
                          </div>

                            <div

                              style={{ height: `${30 + index * 10}%` }}
                            ></div>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                </div>
              </CardContent>
            </Card>
          )}

          {/* Chat Metadata Tab */}
          {activeTab === "chats" && (
            <Card>
              <CardHeader>
                <CardTitle>Chat Metadata</CardTitle>
                <CardDescription>Encrypted communication patterns (content not logged)</CardDescription>
              </CardHeader>
              <CardContent>
                <div className="space-y-6">
                  <div>

                      <table className="w-full">
                        <thead>
                          <tr className="border-b">
                            <th className="text-left py-2">Platform</th>

                            <th className="text-left py-2">Contacts</th>
                            <th className="text-left py-2">Last Activity</th>
                            <th className="text-left py-2">Recent Activity</th>

                        </thead>
                        <tbody>
                          {chatMetadata.map((chat, index) => (

                  <div>

                    <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                      <Card>
                        <CardContent className="pt-4">
                          <div className="text-3xl font-bold text-center">24</div>
                          <p className="text-center text-gray-600">Daily Messages</p>
                        </CardContent>
                      </Card>
                      <Card>
                        <CardContent className="pt-4">
                          <div className="text-3xl font-bold text-center">8</div>
                          <p className="text-center text-gray-600">Active Contacts</p>
                        </CardContent>
                      </Card>
                      <Card>
                        <CardContent className="pt-4">
                          <div className="text-3xl font-bold text-center">3</div>
                          <p className="text-center text-gray-600">New Contacts</p>
                        </CardContent>
                      </Card>
                    </div>
                  </div>
                </div>
              </CardContent>
            </Card>
          )}

          {/* Settings Tab */}
          {activeTab === "settings" && (
            <Card>
              <CardHeader>
                <CardTitle>Settings</CardTitle>
                <CardDescription>Configure monitoring preferences and controls</CardDescription>
              </CardHeader>
              <CardContent className="space-y-6">
                <div>
                  <h3 className="font-medium mb-3">Monitoring Controls</h3>
                  <div className="space-y-4">
                    <div className="flex items-center justify-between">

                      </Button>
                    </div>
                    <div className="flex items-center justify-between">
                      <div>
                        <h4 className="font-medium">Notification Alerts</h4>
                        <p className="text-sm text-gray-600">Receive alerts on your device</p>
                      </div>
                      <Button 
                        onClick={toggleNotifications}
                        variant={notificationsEnabled ? "default" : "secondary"}
                      >
                        {notificationsEnabled ? "Enabled" : "Enable"}
                      </Button>
                    </div>
                    <div className="flex items-center justify-between">
                      <div>
                        <h4 className="font-medium">Location Tracking</h4>
                        <p className="text-sm text-gray-600">Monitor child's location</p>
                      </div>
                      <Button variant="secondary">Enable</Button>
                    </div>
                  </div>
                </div>
                
                <div>
                  <h3 className="font-medium mb-3">Alert Preferences</h3>
                  <div className="space-y-4">
                    <div className="flex items-center justify-between">
                      <div>
                        <h4 className="font-medium">Emotional Alerts</h4>
                        <p className="text-sm text-gray-600">Notify about mood changes</p>
                      </div>
                      <Button variant="default">Enabled</Button>
                    </div>
                    <div className="flex items-center justify-between">
                      <div>
                        <h4 className="font-medium">Usage Alerts</h4>
                        <p className="text-sm text-gray-600">Notify about excessive usage</p>
                      </div>
                      <Button variant="default">Enabled</Button>
                    </div>
                    <div className="flex items-center justify-between">
                      <div>
                        <h4 className="font-medium">Social Alerts</h4>
                        <p className="text-sm text-gray-600">Notify about new contacts</p>
                      </div>
                      <Button variant="secondary">Enable</Button>
                    </div>
                  </div>
                </div>
                
                <div>
                  <h3 className="font-medium mb-3">Device Information</h3>
                  <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                    <div>
                      <Label htmlFor="device-name">Device Name</Label>
                      <Input 
                        id="device-name" 
                        value={deviceInfo.name} 
                        onChange={(e) => setDeviceInfo({...deviceInfo, name: e.target.value})}
                      />
                    </div>
                    <div>
                      <Label htmlFor="phone-number">Phone Number</Label>
                      <Input 
                        id="phone-number" 
                        value={deviceInfo.number} 
                        onChange={(e) => setDeviceInfo({...deviceInfo, number: e.target.value})}
                      />
                    </div>
                  </div>
                </div>
              </CardContent>
            </Card>
          )}
        </main>
      </div>
    </div>
  );
}
